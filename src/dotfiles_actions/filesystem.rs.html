22//! Module that contains interfaces common to directives that use filesystems.
23
24use dotfiles_core::Directive;
25use filesystem::FileSystem;
26
27/// Common trait for all the directives that use a Filesystem
28pub trait FileSystemDirective<'a, F: FileSystem>: Directive<'a> {
29  /// Returns the filesystem instance
30  fn fs(&self) -> &F;
31  /// Returns a mutable reference to the filesystem instance
32  fn mut_fs(&mut self) -> &mut F;
33}
