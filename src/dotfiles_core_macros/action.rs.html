22use proc_macro2::TokenStream;
23use quote::quote;
24use syn::DeriveInput;
25
26pub fn expand_conditional_action(input: DeriveInput) -> TokenStream {
27  let (impl_generics, ty_generics, where_clause) = input.generics.split_for_impl();
28
29  let st_name = input.ident;
30  let lifetimes = input.generics.lifetimes().last();
31
32  quote! {
33    #[automatically_derived]
34    impl #impl_generics dotfiles_core::action::ConditionalAction <#lifetimes> for  #st_name #ty_generics #where_clause{
35      fn skip_in_ci(&self) -> bool {
36        self.skip_in_ci
37      }
38    }
39  }
40}
